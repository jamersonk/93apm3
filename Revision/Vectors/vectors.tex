\section{Vectors}
\subsubsection{The Basics}
\begin{definition}
A vector $\vec{v}$ is a quantity with both direction and magnitude.
\[
\vec{v}=\left(v_x,v_y,v_z\right)
\]
\end{definition}
\begin{definition}
The column form of a vector $v$ is
\[
\vec{v}=  
	\left[ {\begin{array}{c}
    		v_x\\
    		v_y\\
    		v_z\\
  	\end{array} } \right]
\]
\end{definition}
\begin{definition}
The magnitude of a vector $\vec{v}$ is given by
\[
|\vec{p}|=\sqrt{{v_x}^2+{v_y}^2+{v_z}^2}
\]
\end{definition}
\begin{definition}
The unit vector $\hat{v}$ of any vector $\vec{v}$ is
\[
\hat{v}=\frac{\vec{v}}{|\vec{v}|}
\]
\end{definition}
\begin{definition}
The \textbf{scalar product} or dot product of two vectors is
\[
\vec{a}\cot\vec{b}=|\vec{a}||\vec{b}|\cos{\theta}
\]
where $\theta$ is the angle between $\vec{a}$ and $\vec{b}$.
\end{definition}
